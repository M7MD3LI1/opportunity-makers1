# ==========================
# ENVIRONMENT VARIABLES
# ==========================
# Create file: backend/.env
# (DO NOT commit this file to git)

DATABASE_URL=file:./dev.db           # SQLite DB path (or PostgreSQL URL for production)
PORT=5000                            # Backend server port
JWT_SECRET=<your-long-random-secret> # Min 32 characters
ENCRYPTION_KEY=<32-char-key>         # Exactly 32 characters for AES-256
APP_URL=https://your-domain.com      # Frontend URL (used for CORS)
SMTP_HOST=smtp.gmail.com             # Email server host
SMTP_PORT=587                        # Email server port
SMTP_USER=<your-email>               # Email address
SMTP_PASS=<your-app-password>        # Email app password
SMTP_FROM="صناع الفرص <email>"       # Sender display name
UPLOAD_DIR=uploads                   # Directory for uploaded files
GROQ_API_KEY=<your-groq-key>         # From console.groq.com
GEMINI_API_KEY=<your-gemini-key>     # From aistudio.google.com
